systeminit（）函数
{
	打开内部时钟；
	关闭MCO，禁止PA8引脚对外输出时钟频率
	关闭高速外部时钟，关闭时钟监测器，关闭锁相环
	接入外部高速时钟
	设置PLL的输入输出
	清除时钟中断标志位
	根据实际情况选择是否设置硬盘
	setsysclock（）
	{
		setsysclockto72（）
		{
			打开外部高速时钟
			检测外部高速时钟是否在限定的时间内成功打开，根据CR寄存器的第17位HSERDY查看，该位由硬件置一
			根据标志位判断外部高速时钟是否开启，
			如果开启，更新成功开启标志位
			判断成功开启的标志位，如果为真，表示外部高速时钟开启，进入if条件语句
			if
			{
				打开flash存储器的预取缓存区
				将flash的ACR寄存器设置为两个等待态
				然后配置RCC->CFGR寄存器，设置HCLK,PCLK2，PCLK1的时钟输出频率
				配置RCC->CFGR寄存器，设置PLL锁相环，将外部高速时钟作为PLL的输入时钟源，9倍频输出
				打开PLL锁相环，
				用一个空循环来等待PLL锁相环启动成功
				选择PLL为系统时钟源
				用一个while循环来判断和等待系统时钟源切换到PLL
			}
			else
			{
				如果没有成功打开外部高速时钟，在这里填写异常处理程序
			}
		}
	}
	匹配flash和mcu之间的读取速度
}
